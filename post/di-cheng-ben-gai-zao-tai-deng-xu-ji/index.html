低成本改造台灯（续集）

晚上，当你从外面回家，室内一片漆黑，你需要开灯时想起了你的智能台灯。于是你掏出手机，默默关闭数据连接，打开WIFI，连接家里的互联网，然后喊了“嘿，Siri”此时你老妈已经跑去把台灯打开了。

有时我会遇到这种情况，从外面回来手机还没有连到家里的无线网；或者是躺在床上喊完Siri要你先解锁iPhone。这样智障一般的操作还在困扰着我，于是我知道，上篇文章讲到的方法很大程度上还是不够智能。本着懒死的人生信条，再次把那块ESP8266扣了出来，把代码打开。如果，能绕过手机严苛的联网机制那么会不会好一些。

利用家中的小度智能音响

之前写过小度的一些东西，总的来说是个智障，也就只能在闲暇放个歌来听听。但是当我打开了小度的官网看到了DuerOS的开发介绍时又觉得小度音响还是可以再折腾一下的。于是上网查了一下关于ESP接入小度的相关资料，果然有很大收获。于是有了以下代码，把ESP接入DuerOS。代码来自网络，有稍加修改。基于blinker接入小度，所以请现在手机上下载blinker，然后添加设备，把接入码复制下来。

#define BLINKER_WIFI
#define BLINKER_DUEROS_OUTLET
#include <Blinker.h>
char auth[] = "46514c91fc14"; //blinker的接入码
char ssid[] = "Xiaomi_F7CF"; //路由器名称
char pswd[] = "abcd6587771"; //你的路由器密码
bool oState = false;
const int OUT_PIN = D0;//接继电器
#define BUTTON_1 "OUTButton"
BlinkerButton Button1(BUTTON_1);
void duerPowerState(const String & state)
{
  BLINKER_LOG("need set power state: ", state);
  if (state == BLINKER_CMD_ON) {
    digitalWrite(OUT_PIN, HIGH);
    BlinkerDuerOS.powerState("on");
    BlinkerDuerOS.print();
    oState = true;
  }
  else if (state == BLINKER_CMD_OFF) {
    digitalWrite(OUT_PIN, LOW);
    BlinkerDuerOS.powerState("off");
    BlinkerDuerOS.print();
    oState = false;
  }
}
void duerQuery(int32_t queryCode)
{
  BLINKER_LOG("DuerOS Query codes: ", queryCode);
  switch (queryCode)
  {
    case BLINKER_CMD_QUERY_TIME_NUMBER :
      BLINKER_LOG("DuerOS Query time");
      BlinkerDuerOS.time(millis());
      BlinkerDuerOS.print();
      break;
    default :
      BlinkerDuerOS.time(millis());/*读取设备运行时间，只要问小度就可以得知设备工作时长了*/
      BlinkerDuerOS.print();
      break;
  }
}
void dataRead(const String & data)
{
  BLINKER_LOG("Blinker readString: ", data);
  Blinker.vibrate();
  uint32_t BlinkerTime = millis();
  Blinker.print(BlinkerTime);
  Blinker.print("millis", BlinkerTime);
}
void setup()
{
  Serial.begin(115200);
  BLINKER_DEBUG.stream(Serial);
  pinMode(OUT_PIN, OUTPUT);
  digitalWrite(OUT_PIN, LOW);
  Blinker.begin(auth, ssid, pswd);
  Blinker.attachData(dataRead);
  Button1.attach(button1_callback);
  BlinkerDuerOS.attachPowerState(duerPowerState);
  BlinkerDuerOS.attachQuery(duerQuery);
}
void loop()
{
  Blinker.run();
}
void button1_callback(const String & state)
{
  if (state == BLINKER_CMD_BUTTON_TAP)
  {
    if (oState) {
      digitalWrite(OUT_PIN, LOW);
      oState = false;
    }
    else {
      digitalWrite(OUT_PIN, HIGH);
      oState = true;
    }
    Button1.print();
  }
}

代码烧入开发板后，可以现在blinker上看到设备在线情况，然后打开小度音响APP，接入blinker就可以看到设备了，修改名称后就能用接近自然语言的方式唤醒设备。总体来说，绕过了唤醒Siri要求解锁的智障操作。从外面回来也只需要扯着嗓子喊小度就好了。像天猫精灵，小度之类的小众产品的使用体验确实一言难尽。
